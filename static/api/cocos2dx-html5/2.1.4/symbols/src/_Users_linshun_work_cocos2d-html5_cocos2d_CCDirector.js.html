108 cc.GLToClipTransform = function (transformOut) {
109     var projection = new cc.kmMat4();
110     cc.kmGLGetMatrix(cc.KM_GL_PROJECTION, projection);
111 
112     var modelview = new cc.kmMat4();
113     cc.kmGLGetMatrix(cc.KM_GL_MODELVIEW, modelview);
114 
115     cc.kmMat4Multiply(transformOut, projection, modelview);
116 };

260     /**
261      * calculates delta time since last time it was called
262      */
263     calculateDeltaTime:function () {
264         var now = Date.now();
265 
266         // new delta time.
267         if (this._nextDeltaTimeZero) {
268             this._deltaTime = 0;
269             this._nextDeltaTimeZero = false;
270         } else {
271             this._deltaTime = (now - this._lastUpdate) / 1000;
272         }
273 
274         if ((cc.COCOS2D_DEBUG > 0) && (this._deltaTime > 0.2))
275             this._deltaTime = 1 / 60.0;
276 
277         this._lastUpdate = now;

280     /**
281      * <p>
282      *     converts a UIKit coordinate to an OpenGL coordinate<br/>
283      *     Useful to convert (multi) touches coordinates to the current layout (portrait or landscape)
284      * </p>
285      * @param {cc.Point} point
286      * @return {cc.Point}
287      */
288     convertToGL:function (uiPoint) {
289         var transform = new cc.kmMat4();
290         cc.GLToClipTransform(transform);
291 
292         var transformInv = new cc.kmMat4();
293         cc.kmMat4Inverse(transformInv, transform);
294 
295         // Calculate z=0 using -> transform*[0, 0, 0, 1]/w
296         var zClip = transform.mat[14] / transform.mat[15];
297 
298         var glSize = this._openGLView.getDesignResolutionSize();
299         var clipCoord = new cc.kmVec3(2.0 * uiPoint.x / glSize.width - 1.0, 1.0 - 2.0 * uiPoint.y / glSize.height, zClip);
300 
301         var glCoord = new cc.kmVec3();
302         cc.kmVec3TransformCoord(glCoord, clipCoord, transformInv);
303 
304         return cc.p(glCoord.x, glCoord.y);
305     },

307     /**
308      * <p>converts an OpenGL coordinate to a UIKit coordinate<br/>
309      * Useful to convert node points to window points for calls such as glScissor</p>
310      * @param {cc.Point} point
311      * @return {cc.Point}
312      */
313     convertToUI:function (glPoint) {
314         var transform = new cc.kmMat4();
315         cc.GLToClipTransform(transform);
316 
317         var clipCoord = new cc.kmVec3();
318         // Need to calculate the zero depth from the transform.
319         var glCoord = new cc.kmVec3(glPoint.x, glPoint.y, 0.0);
320         cc.kmVec3TransformCoord(clipCoord, glCoord, transform);
321 
322         var glSize = this._openGLView.getDesignResolutionSize();
323         return cc.p(glSize.width * (clipCoord.x * 0.5 + 0.5), glSize.height * (-clipCoord.y * 0.5 + 0.5));
324     },

326     /**
327      *  Draw the scene. This method is called every frame. Don't call it manually.

329     drawScene:function () {
330         // calculate "global" dt
331         this.calculateDeltaTime();
332 
333         //tick before glClear: issue #533
334         if (!this._paused)
335             this._scheduler.update(this._deltaTime);
336 
337         if (cc.renderContextType === cc.CANVAS)
338             this._drawSceneForCanvas();
339         else
340             this._drawSceneForWebGL();
341 
342         this._totalFrames++;
343 
344         // swap buffers
345         /*        if (this._openGLView) {
346          this._openGLView.swapBuffers();
347          }*/
348 
349         if (this._displayStats)
350             this._calculateMPF();
351     },

353     _drawSceneForCanvas:function () {
354         cc.renderContext.clearRect(0, 0, cc.canvas.width, -cc.canvas.height);
355 
356         /* to avoid flickr, nextScene MUST be here: after tick and before draw.
357          XXX: Which bug is this one. It seems that it can't be reproduced with v0.9 */
358         if (this._nextScene) {
359             this.setNextScene();
360         }
361         // draw the scene
362         if (this._runningScene)
363             this._runningScene.visit();
364 
365         // draw the notifications node
366         if (this._notificationNode)
367             this._notificationNode.visit();
368 
369         if (this._displayStats)
370             this._showStats();
371     },
372 
373     _drawSceneForWebGL:function () {
374         var gl = cc.renderContext;
375         gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
376 
377         /* to avoid flickr, nextScene MUST be here: after tick and before draw.
378          XXX: Which bug is this one. It seems that it can't be reproduced with v0.9 */
379         if (this._nextScene)
380             this.setNextScene();
381 
382         cc.kmGLPushMatrix();
383 
384         // draw the scene
385         if (this._runningScene)
386             this._runningScene.visit();
387 
388         // draw the notifications node
389         if (this._notificationNode)
390             this._notificationNode.visit();
391 
392         if (this._displayStats)
393             this._showStats();
394 
395         cc.kmGLPopMatrix();

598             this._nextScene = scene;

709     /**
710      * set next scene
711      */
712     setNextScene:function () {
713         var runningIsTransition = this._runningScene ? this._runningScene instanceof cc.TransitionScene : false;
714 
715         var newIsTransition = this._nextScene ? this._nextScene instanceof cc.TransitionScene : false;
716 
717         // If it is not a transition, call onExit/cleanup
718         if (!newIsTransition) {
719             if (this._runningScene) {
720                 this._runningScene.onExitTransitionDidStart();
721                 this._runningScene.onExit();
722             }
723 
724             // issue #709. the root node (scene) should receive the cleanup message too
725             // otherwise it might be leaked.
726             if (this._sendCleanupToScene && this._runningScene)
727                 this._runningScene.cleanup();
728         }
729 
730         this._runningScene = this._nextScene;
731 
732         this._nextScene = null;
733         if ((!runningIsTransition) && (this._runningScene != null)) {
734             this._runningScene.onEnter();
735             this._runningScene.onEnterTransitionDidFinish();
736         }
737     },

791     /**
792      * Sets an OpenGL projection
793      * @param {Number} projection
794      */
795     setProjection:function (projection) {
796         var size = this._winSizeInPoints;
797 
798         if(cc.renderContextType === cc.WEBGL){
799             if (this._openGLView)
800                 this._openGLView.setViewPortInPoints(0, 0, size.width, size.height);
801 
802             switch (projection) {
803                 case cc.DIRECTOR_PROJECTION_2D:
804                     cc.kmGLMatrixMode(cc.KM_GL_PROJECTION);
805                     cc.kmGLLoadIdentity();
806                     var orthoMatrix = new cc.kmMat4();
807                     cc.kmMat4OrthographicProjection(orthoMatrix, 0, size.width, 0, size.height, -1024, 1024);
808                     cc.kmGLMultMatrix(orthoMatrix);
809                     cc.kmGLMatrixMode(cc.KM_GL_MODELVIEW);
810                     cc.kmGLLoadIdentity();
811                     break;
812                 case cc.DIRECTOR_PROJECTION_3D:
813                     var zeye = this.getZEye();
814                     var matrixPerspective = new cc.kmMat4(), matrixLookup = new cc.kmMat4();
815                     cc.kmGLMatrixMode(cc.KM_GL_PROJECTION);
816                     cc.kmGLLoadIdentity();
817 
818                     // issue #1334
819                     cc.kmMat4PerspectiveProjection(matrixPerspective, 60, size.width / size.height, 0.1, zeye * 2);
820                     // kmMat4PerspectiveProjection( &matrixPerspective, 60, (GLfloat)size.width/size.height, 0.1f, 1500);
821 
822                     cc.kmGLMultMatrix(matrixPerspective);
823 
824                     cc.kmGLMatrixMode(cc.KM_GL_MODELVIEW);
825                     cc.kmGLLoadIdentity();
826                     var eye = cc.kmVec3Fill(null, size.width / 2, size.height / 2, zeye);
827                     var center = cc.kmVec3Fill(null, size.width / 2, size.height / 2, 0.0);
828                     var up = cc.kmVec3Fill(null, 0.0, 1.0, 0.0);
829                     cc.kmMat4LookAt(matrixLookup, eye, center, up);
830                     cc.kmGLMultMatrix(matrixLookup);
831                     break;
832                 case cc.DIRECTOR_PROJECTION_CUSTOM:
833                     if (this._projectionDelegate)
834                         this._projectionDelegate.updateProjection();
835                     break;
836                 default:
837                     cc.log("cocos2d: Director: unrecognized projection");
838                     break;
839             }
840         }
841         this._projection = projection;
842         cc.setProjectionMatrixDirty();
843     },

964     /**
965      * <p>
966      *     Pops out all scenes from the queue until the root scene in the queue. <br/>
967      *     This scene will replace the running one.  <br/>
968      *     The running scene will be deleted. If there are no more scenes in the stack the execution is terminated. <br/>
969      *     ONLY call it if there is a running scene.
970      * </p>
971      */
972     popToRootScene:function () {
973         cc.Assert(this._runningScene != null, "A running Scene is needed");
974         var c = this._scenesStack.length;
975 
976         if (c == 1) {
977             this._scenesStack.pop();
978             this.end();
979         } else {
980             while (c > 1) {
981                 var current = this._scenesStack.pop();
982                 if (current.isRunning()) {
983                     current.onExitTransitionDidStart();
984                     current.onExit();
985                 }
986                 current.cleanup();
987                 c--;
988             }
989             this._nextScene = this._scenesStack[this._scenesStack.length - 1];
990             this._sendCleanupToScene = false;
991         }
992     },
