 47 /**
 48  * @param {Number} x
 49  * @return {Number}
 50  * Constructor
 51  */
 52 cc.NextPOT = function (x) {
 53     x = x - 1;
 54     x = x | (x >> 1);
 55     x = x | (x >> 2);
 56     x = x | (x >> 4);
 57     x = x | (x >> 8);
 58     x = x | (x >> 16);
 59     return x + 1;
 60 };

 62 /**
 63  * cc.RenderTexture is a generic rendering target. To render things into it,<br/>
 64  * simply construct a render target, call begin on it, call visit on any cocos<br/>
 65  * scenes or objects to render them, and call end. For convienience, render texture<br/>
 66  * adds a sprite as it's display child with the results, so you can simply add<br/>
 67  * the render texture to your scene and treat it like any other CocosNode.<br/>
 68  * There are also functions for saving the render texture to disk in PNG or JPG format.
 69  * @class
 70  * @extends cc.Node

167     /**
168      * @param {Number} width
169      * @param {Number} height
170      * @param {cc.IMAGE_FORMAT_JPEG|cc.IMAGE_FORMAT_PNG|cc.IMAGE_FORMAT_RAWDATA} format
171      * @param {Number} depthStencilFormat
172      * @return {Boolean}
173      */
174     initWithWidthAndHeight:function (width, height, format, depthStencilFormat) {
175         if (cc.renderContextType === cc.CANVAS) {
176             this.canvas.width = width || 10;
177             this.canvas.height = height || 10;
178 
179             this.context.translate(0, this.canvas.height);
180             this._sprite = cc.Sprite.createWithTexture(this.canvas);
181 
182             return true;
183         } else {
184             cc.Assert(this._pixelFormat != cc.TEXTURE_2D_PIXEL_FORMAT_A8, "only RGB and RGBA formats are valid for a render texture");
185 
186             var gl = cc.renderContext;
187 
188             width = 0 | (width * cc.CONTENT_SCALE_FACTOR());
189             height = 0 | (height * cc.CONTENT_SCALE_FACTOR());
190 
191             this._oldFBO = gl.getParameter(gl.FRAMEBUFFER_BINDING);
192 
193             // textures must be power of two squared
194             var powW , powH;
195 
196             if (cc.Configuration.getInstance().supportsNPOT()) {
197                 powW = width;
198                 powH = height;
199             } else {
200                 powW = cc.NextPOT(width);
201                 powH = cc.NextPOT(height);
202             }
203 
204             //void *data = malloc(powW * powH * 4);
205             var dataLen = powW * powH * 4;
206             var data = new Uint8Array(dataLen);
207             //memset(data, 0, (int)(powW * powH * 4));
208             for (var i = 0; i < powW * powH * 4; i++)
209                 data[i] = 0;
210 
211             this._pixelFormat = format;
212 
213             this._texture = new cc.Texture2D();
214             if (!this._texture)
215                 return false;
216 
217             this._texture.initWithData(data, this._pixelFormat, powW, powH, cc.size(width, height));
218             //free( data );
219 
220             var oldRBO = gl.getParameter(gl.RENDERBUFFER_BINDING);
221 
222             if (cc.Configuration.getInstance().checkForGLExtension("GL_QCOM")) {
223                 this._textureCopy = new cc.Texture2D();
224                 if (!this._textureCopy) {
225                     return false;
226                 }
227                 this._textureCopy.initWithData(data, this._pixelFormat, powW, powH, cc.size(width, height));
228             }
229 
230             // generate FBO
231             this._fBO = gl.createFramebuffer();
232             gl.bindFramebuffer(gl.FRAMEBUFFER, this._fBO);
233 
234             // associate texture with FBO
235             gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this._texture._webTextureObj, 0);
236 
237             if (this._depthRenderBuffer != 0) {
238                 //create and attach depth buffer
239                 this._depthRenderBuffer = gl.createRenderbuffer();
240                 gl.bindRenderbuffer(gl.RENDERBUFFER, this._depthRenderBuffer);
241                 gl.renderbufferStorage(gl.RENDERBUFFER, depthStencilFormat, powW, powH);
242                 gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this._depthRenderBuffer);
243 
244                 // if depth format is the one with stencil part, bind same render buffer as stencil attachment
245                 //if (depthStencilFormat == gl.DEPTH24_STENCIL8)
246                 //    gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this._depthRenderBuffer);
247             }
248 
249             // check if it worked (probably worth doing :) )
250             cc.Assert(gl.checkFramebufferStatus(gl.FRAMEBUFFER) === gl.FRAMEBUFFER_COMPLETE, "Could not attach texture to framebuffer");
251 
252             this._texture.setAliasTexParameters();
253 
254             this._sprite = cc.Sprite.createWithTexture(this._texture);
255             this._sprite.setScaleY(-1);
256             this._sprite.setBlendFunc(gl.ONE, gl.ONE_MINUS_SRC_ALPHA);
257 
258             gl.bindRenderbuffer(gl.RENDERBUFFER, oldRBO);
259             gl.bindFramebuffer(gl.FRAMEBUFFER, this._oldFBO);
260 
261             // Diabled by default.
262             this._autoDraw = false;
263 
264             // add sprite for backward compatibility
265             this.addChild(this._sprite);
266             return true;
267         }
268     },

270     /**
271      * starts grabbing
272      */
273     begin:function () {
274         if (cc.renderContextType === cc.CANVAS)
275             return;
276 
277         // Save the current matrix
278         cc.kmGLPushMatrix();
279 
280         var texSize = this._texture.getContentSizeInPixels();
281 
282         // Calculate the adjustment ratios based on the old and new projections
283         var size = cc.Director.getInstance().getWinSizeInPixels();
284         var widthRatio = size.width / texSize.width;
285         var heightRatio = size.height / texSize.height;
286 
287         var gl = cc.renderContext;
288 
289         // Adjust the orthographic projection and viewport
290         gl.viewport(0, 0, texSize.width, texSize.height);
291 
292         var orthoMatrix = new cc.kmMat4();
293         cc.kmMat4OrthographicProjection(orthoMatrix, -1.0 / widthRatio, 1.0 / widthRatio,
294             -1.0 / heightRatio, 1.0 / heightRatio, -1, 1);
295         cc.kmGLMultMatrix(orthoMatrix);
296 
297         this._oldFBO = gl.getParameter(gl.FRAMEBUFFER_BINDING);
298         gl.bindFramebuffer(gl.FRAMEBUFFER, this._fBO);//Will direct drawing to the frame buffer created above
299 
300         /*  Certain Qualcomm Andreno gpu's will retain data in memory after a frame buffer switch which corrupts the render to the texture.
301          *   The solution is to clear the frame buffer before rendering to the texture. However, calling glClear has the unintended result of clearing the current texture.
302          *   Create a temporary texture to overcome this. At the end of CCRenderTexture::begin(), switch the attached texture to the second one, call glClear,
303          *   and then switch back to the original texture. This solution is unnecessary for other devices as they don't have the same issue with switching frame buffers.
304          */
305         if (cc.Configuration.getInstance().checkForGLExtension("GL_QCOM")) {
306             // -- bind a temporary texture so we can clear the render buffer without losing our texture
307             gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this._textureCopy._webTextureObj, 0);
308             //cc.CHECK_GL_ERROR_DEBUG();
309             gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
310             gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this._texture._webTextureObj, 0);
311         }
312     },

314     /**
315      * starts rendering to the texture while clearing the texture first.<br/>
316      * This is more efficient then calling -clear first and then -begin
317      * @param {Number} r red 0-1
318      * @param {Number} g green 0-1
319      * @param {Number} b blue 0-1
320      * @param {Number} a alpha 0-1 0 is transparent
321      * @param {Number} depthValue
322      * @param {Number} stencilValue
323      */
324     beginWithClear:function (r, g, b, a, depthValue, stencilValue) {
325         var gl = cc.renderContext;
326         depthValue = depthValue || gl.COLOR_BUFFER_BIT;
327         stencilValue = stencilValue || (gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
328 
329         this._beginWithClear(r, g, b, a, depthValue, stencilValue, (gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT | gl.STENCIL_BUFFER_BIT));
330     },
331 
332     _beginWithClear:function (r, g, b, a, depthValue, stencilValue, flags) {
333         if (cc.renderContextType === cc.CANVAS)
334             return;
335 
336         this.begin();
337 
338         var gl = cc.renderContext;
339 
340         // save clear color
341         var clearColor = [0.0, 0.0, 0.0, 0.0];
342         var depthClearValue = 0.0;
343         var stencilClearValue = 0;
344 
345         if (flags & gl.COLOR_BUFFER_BIT) {
346             clearColor = gl.getParameter(gl.COLOR_CLEAR_VALUE);
347             gl.clearColor(r, g, b, a);
348         }
349 
350         if (flags & gl.DEPTH_BUFFER_BIT) {
351             depthClearValue = gl.getParameter(gl.DEPTH_CLEAR_VALUE);
352             gl.clearDepth(depthValue);
353         }
354 
355         if (flags & gl.STENCIL_BUFFER_BIT) {
356             stencilClearValue = gl.getParameter(gl.STENCIL_CLEAR_VALUE);
357             gl.clearStencil(stencilValue);
358         }
359 
360         gl.clear(flags);
361 
362         // restore
363         if (flags & gl.COLOR_BUFFER_BIT)
364             gl.clearColor(clearColor[0], clearColor[1], clearColor[2], clearColor[3]);
365 
366         if (flags & gl.DEPTH_BUFFER_BIT)
367             gl.clearDepth(depthClearValue);
368 
369         if (flags & gl.STENCIL_BUFFER_BIT)
370             gl.clearStencil(stencilClearValue);
371     },
372 
373     /**
374      * ends grabbing
375      */
376     end:function () {
377         if (cc.renderContextType === cc.CANVAS)
378             return;
379 
380         var gl = cc.renderContext;
381         gl.bindFramebuffer(gl.FRAMEBUFFER, this._oldFBO);
382         cc.kmGLPopMatrix();
383 
384         var director = cc.Director.getInstance();
385 
386         var size = director.getWinSizeInPixels();
387 
388         // restore viewport
389         gl.viewport(0, 0, size.width * cc.CONTENT_SCALE_FACTOR(), size.height * cc.CONTENT_SCALE_FACTOR());
390 
391         // special viewport for 3d projection + retina display
392         if (director.getProjection() == cc.DIRECTOR_PROJECTION_3D && cc.CONTENT_SCALE_FACTOR() != 1) {
393             gl.viewport((-size.width / 2), (-size.height / 2), (size.width * cc.CONTENT_SCALE_FACTOR()), (size.height * cc.CONTENT_SCALE_FACTOR()));
394         }
395 
396         director.setProjection(director.getProjection());
397     },

399     /**
400      * clears the texture with a color
401      * @param {Number|cc.Rect} r red 0-1
402      * @param {Number} g green 0-1
403      * @param {Number} b blue 0-1
404      * @param {Number} a alpha 0-1
405      */
406     clear:function (r, g, b, a) {
407         if (cc.renderContextType === cc.CANVAS) {
408             var rect = r;
409             if (rect)
410                 this.context.clearRect(rect.x, rect.y, rect.width, rect.height);
411             else
412                 this.context.clearRect(0, 0, this.canvas.width, -this.canvas.height);
413         } else {
414             this.beginWithClear(r, g, b, a);
415             this.end();
416         }
417     },

419     /**
420      * clears the texture with a specified depth value
421      * @param {Number} dep
422      */
423     clearDepth:function (depthValue) {
424         if (cc.renderContextType === cc.CANVAS)
425             return;
426 
427         this.begin();
428 
429         var gl = cc.renderContext;
430         //! save old depth value
431         var depthClearValue = gl.getParameter(gl.DEPTH_CLEAR_VALUE);
432 
433         gl.clearDepth(depthValue);
434         gl.clear(gl.DEPTH_BUFFER_BIT);
435 
436         // restore clear color
437         gl.clearDepth(depthClearValue);
438         this.end();
439     },

460     visit:function (ctx) {
461         if (cc.renderContextType === cc.CANVAS) {
462             this._super(ctx);
463             return;
464         }
465         // override visit.
466         // Don't call visit on its children
467         if (!this._visible)
468             return;
469 
470         cc.kmGLPushMatrix();
471 
472         if (this._grid && this._grid.isActive()) {
473             this._grid.beforeDraw();
474             this.transformAncestors();
475         }
476 
477         this.transform(ctx);
478         this._sprite.visit();
479         this.draw(ctx);
480 
481         if (this._grid && this._grid.isActive())
482             this._grid.afterDraw(this);
483 
484         cc.kmGLPopMatrix();
485 
486         this._orderOfArrival = 0;
487     },
488 
489     draw:function (ctx) {
490         if (cc.renderContextType === cc.CANVAS) {
491             this._super(ctx);
492             return;
493         }
494 
495         var gl = cc.renderContext;
496         if (this._autoDraw) {
497             this.begin();
498 
499             if (this._clearFlags) {
500                 var oldClearColor = [0.0, 0.0, 0.0, 0.0];
501                 var oldDepthClearValue = 0.0;
502                 var oldStencilClearValue = 0;
503 
504                 // backup and set
505                 if (this._clearFlags & gl.COLOR_BUFFER_BIT) {
506                     oldClearColor = gl.getParameter(gl.COLOR_CLEAR_VALUE);
507                     gl.clearColor(this._clearColor.r, this._clearColor.g, this._clearColor.b, this._clearColor.a);
508                 }
509 
510                 if (this._clearFlags & gl.DEPTH_BUFFER_BIT) {
511                     oldDepthClearValue = gl.getParameter(gl.DEPTH_CLEAR_VALUE);
512                     gl.clearDepth(this._clearDepth);
513                 }
514 
515                 if (this._clearFlags & gl.STENCIL_BUFFER_BIT) {
516                     oldStencilClearValue = gl.getParameter(gl.STENCIL_CLEAR_VALUE);
517                     gl.clearStencil(this._clearStencil);
518                 }
519 
520                 // clear
521                 gl.clear(this._clearFlags);
522 
523                 // restore
524                 if (this._clearFlags & gl.COLOR_BUFFER_BIT)
525                     gl.clearColor(oldClearColor[0], oldClearColor[1], oldClearColor[2], oldClearColor[3]);
526 
527                 if (this._clearFlags & gl.DEPTH_BUFFER_BIT)
528                     gl.clearDepth(oldDepthClearValue);
529 
530                 if (this._clearFlags & gl.STENCIL_BUFFER_BIT)
531                     gl.clearStencil(oldStencilClearValue);
532             }
533 
534             //! make sure all children are drawn
535             this.sortAllChildren();
536 
537             for (var i = 0; i < this._children.length; i++) {
538                 var getChild = this._children[i];
539                 if (getChild != this._sprite)
540                     getChild.visit();
541             }
542 
543             this.end();
544         }
545     },

547     /**
548      * creates a new CCImage from with the texture's data. Caller is responsible for releasing it by calling delete.
549      * @return {cc.Image}
550      */
551     newCCImage:function (flipImage) {
552         if(flipImage === null)
553             flipImage = true;
554         cc.Assert(this._pixelFormat == cc.TEXTURE_2D_PIXEL_FORMAT_RGBA8888, "only RGBA8888 can be saved as image");
555 
556         if (!this._texture)
557             return null;
558 
559         var size = this._texture.getContentSizeInPixels();
560 
561         // to get the image size to save
562         //        if the saving image domain exeeds the buffer texture domain,
563         //        it should be cut
564         var nSavedBufferWidth = size.width;
565         var nSavedBufferHeight = size.height;
566 
567         var pImage = new cc.Image();
568         var gl = cc.renderContext;
569 
570         var pBuffer = new Uint8Array(nSavedBufferWidth * nSavedBufferHeight * 4);
571         if (!(pBuffer))
572             return pImage;
573 
574         var pTempData = new Uint8Array(nSavedBufferWidth * nSavedBufferHeight * 4);
575         if (!(pTempData))
576             return null;
577 
578         this.begin();
579         gl.pixelStorei(gl.PACK_ALIGNMENT, 1);
580         gl.readPixels(0, 0, nSavedBufferWidth, nSavedBufferHeight, gl.RGBA, gl.UNSIGNED_BYTE, pTempData);
581         this.end();
582 
583         // to get the actual texture data
584         // #640 the image read from rendertexture is upseted
585         for (var i = 0; i < nSavedBufferHeight; ++i) {
586             this._memcpy(pBuffer, i * nSavedBufferWidth * 4,
587                 pTempData, (nSavedBufferHeight - i - 1) * nSavedBufferWidth * 4,
588                 nSavedBufferWidth * 4);
589         }
590         pImage.initWithImageData(pBuffer, nSavedBufferWidth * nSavedBufferHeight * 4, cc.FMT_RAWDATA, nSavedBufferWidth, nSavedBufferHeight, 8);
591 
592         pBuffer = null;
593         pTempData = null;
594         return pImage;
595     },
596 
597     _memcpy:function (destArr, destIndex, srcArr, srcIndex, size) {
598         for (var i = 0; i < size; i++) {
599             destArr[destIndex + i] = srcArr[srcIndex + i];
600         }

693  * creates a RenderTexture object with width and height in Points and a pixel format, only RGB and RGBA formats are valid
694  * @param {Number} width
695  * @param {Number} height
696  * @param {cc.IMAGE_FORMAT_JPEG|cc.IMAGE_FORMAT_PNG|cc.IMAGE_FORMAT_RAWDATA} format
697  * @param {Number} depthStencilFormat
698  * @return {cc.RenderTexture}
699  * @example
700  * // Example
701  * var rt = cc.RenderTexture.create()
702  */
703 cc.RenderTexture.create = function (width, height, format, depthStencilFormat) {
704     format = format || cc.TEXTURE_2D_PIXEL_FORMAT_RGBA8888;
705     depthStencilFormat = depthStencilFormat || 0;
706 
707     var ret = new cc.RenderTexture();
708     if (ret && ret.initWithWidthAndHeight(width, height, format, depthStencilFormat))
709         return ret;
710     return null;
711 };
