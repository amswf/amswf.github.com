 27 /**
 28  * <p>
 29  *     CCParticleSystemQuad is a subclass of CCParticleSystem<br/>
 30  *     <br/>
 31  *     It includes all the features of ParticleSystem.<br/>
 32  *     <br/>
 33  *     Special features and Limitations:<br/>
 34  *      - Particle size can be any float number. <br/>
 35  *      - The system can be scaled <br/>
 36  *      - The particles can be rotated     <br/>
 37  *      - It supports subrects   <br/>
 38  *      - It supports batched rendering since 1.1<br/>
 39  * </p>
 40  * @class
 41  * @extends cc.ParticleSystem
 42  * @example
 43  * //create a particle system
 44  *   this._emitter = new cc.ParticleSystemQuad();
 45  *   this._emitter.initWithTotalParticles(150);
 46  */

 74     /**
 75      * initialices the indices for the vertices
 76      */
 77     initIndices:function () {
 78         for (var i = 0; i < this._totalParticles; ++i) {
 79             var i6 = i * 6;
 80             var i4 = i * 4;
 81             this._indices[i6 + 0] = i4 + 0;
 82             this._indices[i6 + 1] = i4 + 1;
 83             this._indices[i6 + 2] = i4 + 2;
 84 
 85             this._indices[i6 + 5] = i4 + 1;
 86             this._indices[i6 + 4] = i4 + 2;
 87             this._indices[i6 + 3] = i4 + 3;
 88         }

 91     /**
 92      * <p> initilizes the texture with a rectangle measured Points<br/>
 93      * pointRect should be in Texture coordinates, not pixel coordinates
 94      * </p>
 95      * @param {cc.Rect} pointRect
 96      */
 97     initTexCoordsWithRect:function (pointRect) {
 98         var scaleFactor = cc.CONTENT_SCALE_FACTOR();
 99         // convert to pixels coords
100         var rect = cc.rect(
101             pointRect.x * scaleFactor,
102             pointRect.y * scaleFactor,
103             pointRect.width * scaleFactor,
104             pointRect.height * scaleFactor);
105 
106         var wide = pointRect.width;
107         var high = pointRect.height;
108 
109         if (this._texture) {
110             if ((this._texture instanceof HTMLImageElement) || (this._texture instanceof HTMLCanvasElement)) {
111                 wide = this._texture.width;
112                 high = this._texture.height;
113             } else {
114                 wide = this._texture.getPixelsWide();
115                 high = this._texture.getPixelsHigh();
116             }
117         }
118 
119         if(cc.renderContextType === cc.CANVAS)
120             return;
121 
122         var left, bottom, right, top;
123         if (cc.FIX_ARTIFACTS_BY_STRECHING_TEXEL) {
124             left = (rect.x * 2 + 1) / (wide * 2);
125             bottom = (rect.y * 2 + 1) / (high * 2);
126             right = left + (rect.width * 2 - 2) / (wide * 2);
127             top = bottom + (rect.height * 2 - 2) / (high * 2);
128         } else {
129             left = rect.x / wide;
130             bottom = rect.y / high;
131             right = left + rect.width / wide;
132             top = bottom + rect.height / high;
133         }
134 
135         // Important. Texture in cocos2d are inverted, so the Y component should be inverted
136         var temp = top;
137         top = bottom;
138         bottom = temp;
139 
140         var quads;
141         var start = 0, end = 0;
142         if (this._batchNode) {
143             quads = this._batchNode.getTextureAtlas().getQuads();
144             start = this._atlasIndex;
145             end = this._atlasIndex + this._totalParticles;
146         } else {
147             quads = this._quads;
148             start = 0;
149             end = this._totalParticles;
150         }
151 
152         for (var i = start; i < end; i++) {
153             if (!quads[i])
154                 quads[i] = cc.V3F_C4B_T2F_QuadZero();
155 
156             // bottom-left vertex:
157             quads[i].bl.texCoords.u = left;
158             quads[i].bl.texCoords.v = bottom;
159             // bottom-right vertex:
160             quads[i].br.texCoords.u = right;
161             quads[i].br.texCoords.v = bottom;
162             // top-left vertex:
163             quads[i].tl.texCoords.u = left;
164             quads[i].tl.texCoords.v = top;
165             // top-right vertex:
166             quads[i].tr.texCoords.u = right;
167             quads[i].tr.texCoords.v = top;
168         }
169     },

229                 // gravity

233                 // speed
234                 retParticle.modeA.speed = this.modeA.speed;
235                 retParticle.modeA.speedVar = this.modeA.speedVar;
236 
237                 // radial acceleration
238                 retParticle.modeA.radialAccel = this.modeA.radialAccel;
239 
240                 retParticle.modeA.radialAccelVar = this.modeA.radialAccelVar;
241 
242                 // tangential acceleration
243                 retParticle.modeA.tangentialAccel = this.modeA.tangentialAccel;
244 
245                 retParticle.modeA.tangentialAccelVar = this.modeA.tangentialAccelVar;
246             } else if (this._emitterMode == cc.PARTICLE_MODE_RADIUS) {
247                 // or Mode B: radius movement
248                 retParticle.modeB.startRadius = this.modeB.startRadius;
249                 retParticle.modeB.startRadiusVar = this.modeB.startRadiusVar;
250                 retParticle.modeB.endRadius = this.modeB.endRadius;

252                 retParticle.modeB.rotatePerSecond = this.modeB.rotatePerSecond;
253                 retParticle.modeB.rotatePerSecondVar = this.modeB.rotatePerSecondVar;
254             }
255 
256             // life span
257             retParticle._life = this._life;
258             retParticle._lifeVar = this._lifeVar;
259 
260             // emission Rate
261             retParticle._emissionRate = this._emissionRate;
262 
263             //don't get the internal texture if a batchNode is used
264             if (!this._batchNode) {
265                 // Set a compatible default for the alpha transfer
266                 retParticle._opacityModifyRGB = this._opacityModifyRGB;
267 
268                 // texture
269                 if (this._texture instanceof cc.Texture2D)
270                     retParticle._texture = this._texture;
271                 else
272                     retParticle._texture = this._texture;
273             }
274         }
275         return retParticle;

293     /**
294      *  Sets a new texture with a rect. The rect is in Points.
295      * @param {cc.Texture2D} texture
296      * @param {cc.Rect} rect
297      */
298     setTextureWithRect:function (texture, rect) {
299         if (cc.renderContextType === cc.WEBGL) {
300             // Only update the texture if is different from the current one
301             if (!this._texture || texture._webTextureObj != this._texture._webTextureObj)
302                 cc.ParticleSystem.prototype.setTexture.call(this, texture);
303             this._pointRect = rect;
304             this.initTexCoordsWithRect(rect);
305         } else {
306             if (!this._texture || texture != this._texture)
307                 cc.ParticleSystem.prototype.setTexture.call(this, texture);
308             this._pointRect = rect;
309             this.initTexCoordsWithRect(rect);
310         }

315     /**
316      * Initializes a system with a fixed number of particles
317      * @override
318      * @param {Number} numberOfParticles
319      * @return {Boolean}
320      */
321     initWithTotalParticles:function (numberOfParticles) {
322         // base initialization
323         if (cc.ParticleSystem.prototype.initWithTotalParticles.call(this, numberOfParticles)) {
324             if (cc.renderContextType === cc.CANVAS)
325                 return true;
326 
327             // allocating data space
328             if (!this._allocMemory())
329                 return false;
330 
331             this.initIndices();
332             //if (cc.TEXTURE_ATLAS_USE_VAO)
333             //    this._setupVBOandVAO();
334             //else
335             this._setupVBO();
336 
337             this.setShaderProgram(cc.ShaderCache.getInstance().programForKey(cc.SHADER_POSITION_TEXTURECOLOR));
338             return true;
339         }
340         return false;
341     },

343     /**
344      * set Texture of Particle System
345      * @override
346      * @param {HTMLImageElement|HTMLCanvasElement|cc.Texture2D} texture
347      * @param {Boolean} isCallSuper is direct call super method
348      */
349     setTexture:function (texture, isCallSuper) {
350         if (isCallSuper != null && isCallSuper === true) {
351             cc.ParticleSystem.prototype.setTexture.call(this, texture);
352             return;
353         }
354         var size = null;
355         if ((texture instanceof HTMLImageElement) || (texture instanceof HTMLCanvasElement))
356             size = cc.size(texture.width, texture.height);
357         else
358             size = texture.getContentSize();
359 
360         this.setTextureWithRect(texture, cc.rect(0, 0, size.width, size.height));
361     },

363     /**
364      * update particle's quad
365      * @override
366      * @param {cc.Particle} particle
367      * @param {cc.Point} newPosition
368      */
369     updateQuadWithParticle:function (particle, newPosition) {
370         var quad = null;
371         if (this._batchNode) {
372             var batchQuads = this._batchNode.getTextureAtlas().getQuads();
373             quad = batchQuads[this._atlasIndex + particle.atlasIndex];
374             this._batchNode.getTextureAtlas()._dirty = true;
375 
376         } else
377             quad = this._quads[this._particleIdx];
378 
379         var r, g, b, a;
380         if(this._opacityModifyRGB){
381             r = 0 | (particle.color.r * particle.color.a * 255);
382             g = 0 | (particle.color.g * particle.color.a * 255);
383             b = 0 | (particle.color.b * particle.color.a * 255);
384             a = 0 | (particle.color.a * 255);
385         }else{
386             r = 0 | (particle.color.r * 255);
387             g = 0 | (particle.color.g * 255);
388             b = 0 | (particle.color.b * 255);
389             a = 0 | (particle.color.a * 255);
390         }
391 
392         quad.bl.colors.r = r;
393         quad.bl.colors.g = g;
394         quad.bl.colors.b = b;
395         quad.bl.colors.a = a;
396 
397         quad.br.colors.r = r;
398         quad.br.colors.g = g;
399         quad.br.colors.b = b;
400         quad.br.colors.a = a;
401 
402         quad.tl.colors.r = r;
403         quad.tl.colors.g = g;
404         quad.tl.colors.b = b;
405         quad.tl.colors.a = a;
406 
407         quad.tr.colors.r = r;
408         quad.tr.colors.g = g;
409         quad.tr.colors.b = b;
410         quad.tr.colors.a = a;
411 
412         // vertices
413         var size_2 = particle.size / 2;
414         if (particle.rotation) {
415             var x1 = -size_2;
416             var y1 = -size_2;
417 
418             var x2 = size_2;
419             var y2 = size_2;
420             var x = newPosition.x;
421             var y = newPosition.y;
422 
423             var rad = -cc.DEGREES_TO_RADIANS(particle.rotation);
424             var cr = Math.cos(rad);
425             var sr = Math.sin(rad);
426             var ax = x1 * cr - y1 * sr + x;
427             var ay = x1 * sr + y1 * cr + y;
428             var bx = x2 * cr - y1 * sr + x;
429             var by = x2 * sr + y1 * cr + y;
430             var cx = x2 * cr - y2 * sr + x;
431             var cy = x2 * sr + y2 * cr + y;
432             var dx = x1 * cr - y2 * sr + x;
433             var dy = x1 * sr + y2 * cr + y;
434 
435             // bottom-left
436             quad.bl.vertices.x = ax;
437             quad.bl.vertices.y = ay;
438 
439             // bottom-right vertex:
440             quad.br.vertices.x = bx;
441             quad.br.vertices.y = by;
442 
443             // top-left vertex:
444             quad.tl.vertices.x = dx;
445             quad.tl.vertices.y = dy;
446 
447             // top-right vertex:
448             quad.tr.vertices.x = cx;
449             quad.tr.vertices.y = cy;
450         } else {
451             // bottom-left vertex:
452             quad.bl.vertices.x = newPosition.x - size_2;
453             quad.bl.vertices.y = newPosition.y - size_2;
454 
455             // bottom-right vertex:
456             quad.br.vertices.x = newPosition.x + size_2;
457             quad.br.vertices.y = newPosition.y - size_2;
458 
459             // top-left vertex:
460             quad.tl.vertices.x = newPosition.x - size_2;
461             quad.tl.vertices.y = newPosition.y + size_2;
462 
463             // top-right vertex:
464             quad.tr.vertices.x = newPosition.x + size_2;
465             quad.tr.vertices.y = newPosition.y + size_2;
466         }
467     },

493     /**
494      * draw particle
495      * @param {CanvasContext} ctx CanvasContext
496      * @override
497      */
498     draw:function (ctx) {
499         cc.Assert(!this._batchNode, "draw should not be called when added to a particleBatchNode");
500 
501         if (cc.renderContextType === cc.CANVAS)
502             this._drawForCanvas(ctx);
503         else
504             this._drawForWebGL(ctx);
505 
506         cc.g_NumberOfDraws++;
507     },

509     _drawForCanvas:function (ctx) {
510         var context = ctx || cc.renderContext;
511         context.save();
512         if (this.isBlendAdditive())
513             context.globalCompositeOperation = 'lighter';
514         else
515             context.globalCompositeOperation = 'source-over';
516 
517         for (var i = 0; i < this._particleCount; i++) {
518             var particle = this._particles[i];
519             var lpx = (0 | (particle.size * 0.5));
520 
521             if (this._drawMode == cc.PARTICLE_TEXTURE_MODE) {
522 
523                 var drawTexture = this.getTexture();
524 
525                 // Delay drawing until the texture is fully loaded by the browser
526                 if (!drawTexture.width || !drawTexture.height)
527                     continue;
528 
529                 context.save();
530                 context.globalAlpha = particle.color.a;
531                 context.translate((0 | particle.drawPos.x), -(0 | particle.drawPos.y));
532 
533                 var size = Math.floor(particle.size / 4) * 4;
534                 var w = this._pointRect.width;
535                 var h = this._pointRect.height;
536 
537                 context.scale(
538                     Math.max((1 / w) * size, 0.000001),
539                     Math.max((1 / h) * size, 0.000001)
540                 );
541 
542 
543                 if (particle.rotation)
544                     context.rotate(cc.DEGREES_TO_RADIANS(particle.rotation));
545 
546                 context.translate(-(0 | (w / 2)), -(0 | (h / 2)));
547                 if (particle.isChangeColor) {
548 
549                     var cacheTextureForColor = cc.TextureCache.getInstance().getTextureColors(drawTexture);
550                     if (cacheTextureForColor) {
551                         // Create another cache for the tinted version
552                         // This speeds up things by a fair bit
553                         if (!cacheTextureForColor.tintCache) {
554                             cacheTextureForColor.tintCache = document.createElement('canvas');
555                             cacheTextureForColor.tintCache.width = drawTexture.width;
556                             cacheTextureForColor.tintCache.height = drawTexture.height;
557                         }
558                         cc.generateTintImage(drawTexture, cacheTextureForColor, particle.color, this._pointRect, cacheTextureForColor.tintCache);
559                         drawTexture = cacheTextureForColor.tintCache;
560                     }
561                 }
562 
563                 context.drawImage(drawTexture, 0, 0);
564                 context.restore();
565 
566             } else {
567                 context.save();
568                 context.globalAlpha = particle.color.a;
569 
570                 context.translate(0 | particle.drawPos.x, -(0 | particle.drawPos.y));
571 
572                 if (this._shapeType == cc.PARTICLE_STAR_SHAPE) {
573                     if (particle.rotation)
574                         context.rotate(cc.DEGREES_TO_RADIANS(particle.rotation));
575                     cc.drawingUtil.drawStar(context, lpx, particle.color);
576                 } else
577                     cc.drawingUtil.drawColorBall(context, lpx, particle.color);
578                 context.restore();
579             }
580         }
581         context.restore();
582     },

584     _drawForWebGL:function (ctx) {
585         if(!this._texture)
586             return;
587 
588         var gl = ctx || cc.renderContext;
589 
590         this._shaderProgram.use();
591         this._shaderProgram.setUniformForModelViewAndProjectionMatrixWithMat4();
592 
593         cc.glBindTexture2D(this._texture);
594         cc.glBlendFuncForParticle(this._blendFunc.src, this._blendFunc.dst);
595 
596         //cc.Assert(this._particleIdx == this._particleCount, "Abnormal error in particle quad");
597 
598         //
599         // Using VBO without VAO
600         //
601         cc.glEnableVertexAttribs(cc.VERTEX_ATTRIB_FLAG_POSCOLORTEX);
602 
603         gl.bindBuffer(gl.ARRAY_BUFFER, this._buffersVBO[0]);
604         gl.vertexAttribPointer(cc.VERTEX_ATTRIB_POSITION, 3, gl.FLOAT, false, 24, 0);               // vertices
605         gl.vertexAttribPointer(cc.VERTEX_ATTRIB_COLOR, 4, gl.UNSIGNED_BYTE, true, 24, 12);          // colors
606         gl.vertexAttribPointer(cc.VERTEX_ATTRIB_TEX_COORDS, 2, gl.FLOAT, false, 24, 16);            // tex coords
607 
608         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this._buffersVBO[1]);
609         gl.drawElements(gl.TRIANGLES, this._particleIdx * 6, gl.UNSIGNED_SHORT, 0);
610     },

640     setTotalParticles:function (tp) {
641         if (cc.renderContextType === cc.CANVAS){
642             this._totalParticles = (tp < 200) ? tp : 200;
643             return;
644         }
645 
646         // If we are setting the total numer of particles to a number higher
647         // than what is allocated, we need to allocate new arrays
648         if (tp > this._allocatedParticles) {
649             var quadSize = cc.V3F_C4B_T2F_Quad.BYTES_PER_ELEMENT;
650             // Allocate new memory
651             this._indices = new Uint16Array(tp * 6);
652             this._quadsArrayBuffer = new ArrayBuffer(tp * quadSize);
653             //TODO need fix
654             // Assign pointers
655             this._particles = [];
656             this._quads = [];
657             for (var j = 0; j < tp; j++) {
658                 this._particles[j] = new cc.Particle();
659                 this._quads[j] = new cc.V3F_C4B_T2F_Quad(null, null, null, null, this._quadsArrayBuffer, j * quadSize);
660             }
661             this._allocatedParticles = tp;
662             this._totalParticles = tp;
663 
664             // Init particles
665             if (this._batchNode) {
666                 for (var i = 0; i < this._totalParticles; i++)
667                     this._particles[i].atlasIndex = i;
668             }
669             this.initIndices();
670             //if (cc.TEXTURE_ATLAS_USE_VAO)
671             //    this._setupVBOandVAO();
672             //else
673             this._setupVBO();
674 
675             //set the texture coord
676             if(this._texture){
677                 var size = this._texture.getContentSize();
678                 this.initTexCoordsWithRect(cc.rect(0, 0, size.width, size.height));
679             }
680         } else
681             this._totalParticles = tp;
682         this.resetSystem();
683     },

735     _setupVBO:function () {
736         if (cc.renderContextType == cc.CANVAS)
737             return;
738 
739         var gl = cc.renderContext;
740 
741         //gl.deleteBuffer(this._buffersVBO[0]);
742         this._buffersVBO[0] = gl.createBuffer();
743         gl.bindBuffer(gl.ARRAY_BUFFER, this._buffersVBO[0]);
744         gl.bufferData(gl.ARRAY_BUFFER, this._quadsArrayBuffer, gl.DYNAMIC_DRAW);
745 
746         this._buffersVBO[1] = gl.createBuffer();
747         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this._buffersVBO[1]);
748         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, this._indices, gl.STATIC_DRAW);
749 
750         //cc.CHECK_GL_ERROR_DEBUG();
751     },
752 
753     _allocMemory:function () {
754         if (cc.renderContextType === cc.CANVAS)
755             return;
756 
757         //cc.Assert((!this._quads && !this._indices), "Memory already allocated");
758         cc.Assert(!this._batchNode, "Memory should not be allocated when not using batchNode");
759 
760         var totalParticles = this._totalParticles;
761         this._quads = [];
762         this._indices = new Uint16Array(totalParticles * 6);
763         this._quadsArrayBuffer = new ArrayBuffer(cc.V3F_C4B_T2F_Quad.BYTES_PER_ELEMENT * totalParticles);
764 
765         for (var i = 0; i < totalParticles; i++)
766             this._quads[i] = new cc.V3F_C4B_T2F_Quad(null, null, null, null, this._quadsArrayBuffer, i * cc.V3F_C4B_T2F_Quad.BYTES_PER_ELEMENT);
767         if (!this._quads || !this._indices) {
768             cc.log("cocos2d: Particle system: not enough memory");
769             return false;
770         }
771         return true;
772     }

775 /**
776  * <p>
777  *   creates an initializes a CCParticleSystemQuad from a plist file.<br/>
778  *   This plist files can be creted manually or with Particle Designer:<br/>
779  *   http://particledesigner.71squared.com/<br/>
780  * </p>
781  * @param {String|Number} pListFile
782  * @return {cc.ParticleSystem}
783  * @example
784  *  //creates an initializes a CCParticleSystemQuad from a plist file.
785  *  var system = cc.ParticleSystemQuad.create("Images/SpinningPeas.plist");
786  */
787 cc.ParticleSystemQuad.create = function (pListFile) {
788     var ret = new cc.ParticleSystemQuad();
789     if (!pListFile || typeof(pListFile) === "number") {
790         var ton = pListFile || 100;
791         ret.setDrawMode(cc.PARTICLE_TEXTURE_MODE);
792         ret.initWithTotalParticles(ton);
793         return ret;
794     }
795 
796     if (ret && ret.initWithFile(pListFile))
797         return ret;
798     return null;
799 };
800 
801 cc.ParticleSystemQuad.createWithTotalParticles = function (numberOfParticles) {
802     return cc.ParticleSystemQuad.create(numberOfParticles);
803 };
