 27 /**
 28  * create a webgl context
 29  * @param {HTMLCanvasElement} canvas
 30  * @param {Object} opt_attribs
 31  * @return {WebGLRenderingContext}
 32  */
 33 cc.create3DContext = function (canvas, opt_attribs) {
 34     var names = ["webgl", "experimental-webgl", "webkit-3d", "moz-webgl"];
 35     var context = null;
 36     for (var ii = 0; ii < names.length; ++ii) {
 37         try {
 38             context = canvas.getContext(names[ii], opt_attribs);
 39         } catch (e) {
 40         }
 41         if (context) {
 42             break;
 43         }
 44     }
 45     return context;
 46 };

 48 /**
 49  * Browser detection, based on mootools<br/>
 50  * platform will print out win32, mac, etc<br/>
 51  * type is the browser type, chrome, firefox etc
 52  * @type {Object}
 53  */
 54 cc.Browser = {};
 55 (function () {
 56     cc.Browser.ua = navigator.userAgent.toLowerCase();
 57     cc.Browser.platform = navigator.platform.toLowerCase();
 58     cc.Browser.UA = cc.Browser.ua.match(/(opera|ie|firefox|chrome|version)[\s\/:]([\w\d\.]+)?.*?(safari|version[\s\/:]([\w\d\.]+)|$)/) || [null, 'unknown', 0];
 59     cc.Browser.mode = cc.Browser.UA[1] == 'ie' && document.documentMode;
 60     cc.Browser.type = (cc.Browser.UA[1] == 'version') ? cc.Browser.UA[3] : cc.Browser.UA[1];
 61     cc.Browser.isMobile = (cc.Browser.ua.indexOf('mobile') != -1 || cc.Browser.ua.indexOf('android') != -1);
 62 
 63     if (!document["ccConfig"])
 64         document["ccConfig"] = {};
 65 
 66     var c = document["ccConfig"];
 67     // check supportWebGL item
 68     cc._userRenderMode = parseInt(c["renderMode"]) || 0;
 69 
 70     if(cc._userRenderMode === 1) {
 71          //canvas only
 72         cc.Browser.supportWebGL = false;
 73     } else{
 74         // WebGL first
 75         cc.Browser.supportWebGL = !(window.WebGLRenderingContext == null);
 76         var tempCanvas = document.createElement("Canvas");
 77         var tempContext = cc.create3DContext(tempCanvas, {'stencil':true, 'preserveDrawingBuffer':true });
 78         cc.Browser.supportWebGL = !(tempContext == null)
 79     }
 80     if(cc._userRenderMode === 2 && !cc.Browser.supportWebGL){
 81         // WebGL render only, but browser doesn't support WebGL.
 82         cc.__renderDoesnotSupport = true;
 83     }
 84 
 85     // check if browser supports Web Audio
 86     cc.Browser.supportWebAudio = (function(){
 87         // check Web Audio's context
 88         try {
 89             var ctx = new (window.AudioContext || window.webkitAudioContext || window.mozAudioContext)();
 90             return ctx ? true : false;
 91         } catch (e) {
 92             return false;
 93         }
 94     })();
 95 })();
 96 
 97 cc.RenderDoesnotSupport = function(){
 98     if(cc.__renderDoesnotSupport === "undefined")
 99         return false;
100     return cc.__renderDoesnotSupport;
101 };

104 /**
105  * the dollar sign, classic like jquery, this selector add extra methods to HTMLElement without touching its prototype</br>
106  * it is also chainable like jquery
107  * @param {HTMLElement|String} x pass in a css selector in string or the whole HTMLElement
108  * @class
109  * @return {cc.$}
110  */
111 cc.$ = function (x) {
112     /** @lends cc.$# */
113     var parent = (this == cc) ? document : this;
114 
115     /**
116      * @type {HTMLElement}
117      */
118     var el = (x instanceof HTMLElement) ? x : parent.querySelector(x);
119 
120     if (el) {
121         /**
122          * find and return the child wth css selector (same as jquery.find)
123          * @param {HTMLElement|String} x pass in a css selector in string or the whole HTMLElement
124          * @return {cc.$}
125          */
126         el.find = el.find || cc.$;
127         /**
128          * check if a DOMNode has a specific class
129          * @param {String} cls
130          * @return {Boolean}
131          */
132         el.hasClass = el.hasClass || function (cls) {
133             return this.className.match(new RegExp('(\\s|^)' + cls + '(\\s|$)'));
134         };
135         /**
136          * add a class to a DOMNode, returns self to allow chaining
137          * @param {String} cls
138          * @return {cc.$}
139          */
140         el.addClass = el.addClass || function (cls) {
141             if (!this.hasClass(cls)) {
142                 if (this.className) {
143                     this.className += " ";
144                 }
145                 this.className += cls;
146             }
147             return this;
148         };
149         /**
150          * remove a specific class from a DOMNode, returns self to allow chaining
151          * @param {String} cls
152          * @return {cc.$}
153          */
154         el.removeClass = el.removeClass || function (cls) {
155             if (this.hasClass(cls)) {
156                 this.className = this.className.replace(cls, '');
157             }
158             return this;
159         };
160         /**
161          * detach it self from parent
162          * @function
163          */
164         el.remove = el.remove || function () {
165             if (this.parentNode)
166                 this.parentNode.removeChild(this);
167             return this;
168         };
169 
170         /**
171          * add to another element as a child
172          * @param {HTMLElement|cc.$} x
173          * @return {cc.$}
174          */
175         el.appendTo = el.appendTo || function (x) {
176             x.appendChild(this);
177             return this;
178         };
179 
180         /**
181          * add to another element as a child and place on the top of the children list
182          * @param {HTMLElement|cc.$} x
183          * @return {cc.$}
184          */
185         el.prependTo = el.prependTo || function (x) {
186             ( x.childNodes[0]) ? x.insertBefore(this, x.childNodes[0]) : x.appendChild(this);
187             return this;
188         };
189 
190         /**
191          * helper function for updating the css transform
192          * @return {cc.$}
193          */
194         el.transforms = el.transforms || function () {
195             this.style[cc.$.trans] = cc.$.translate(this.position) + cc.$.rotate(this.rotation) + cc.$.scale(this.scale) + cc.$.skew(this.skew);
196             return this;
197         };
198 
199         el.position = el.position || {x:0, y:0};
200         el.rotation = el.rotation || 0;
201         el.scale = el.scale || {x:1, y:1};
202         el.skew = el.skew || {x:0, y:0};
203 
204         /**
205          * move the element
206          * @param {Number} x in pixel
207          * @param {Number} y in pixel
208          * @return {cc.$}
209          */
210         el.translates = function (x, y) {
211             this.position.x = x;
212             this.position.y = y;
213             this.transforms();
214             return this
215         };
216 
217         /**
218          * rotate the element
219          * @param {Number} x in degrees
220          * @return {cc.$}
221          */
222         el.rotate = function (x) {
223             this.rotation = x;
224             this.transforms();
225             return this
226         };
227 
228         /**
229          * resize the element
230          * @param {Number} x
231          * @param {Number} y
232          * @return {cc.$}
233          */
234         el.resize = function (x, y) {
235             this.scale.x = x;
236             this.scale.y = y;
237             this.transforms();
238             return this
239         };
240 
241         /**
242          * skews the element
243          * @param {Number} x in degrees
244          * @param {Number} y
245          * @return {cc.$}
246          */
247         el.setSkew = function (x, y) {
248             this.skew.x = x;
249             this.skew.y = y;
250             this.transforms();
251             return this
252         };
253     }
254     return el;
255 };
256 //getting the prefix and css3 3d support
257 switch (cc.Browser.type) {
258     case "firefox":
259         cc.$.pfx = "Moz";
260         cc.$.hd = true;
261         break;
262     case "chrome":
263     case "safari":
264         cc.$.pfx = "webkit";
265         cc.$.hd = true;
266         break;
267     case "opera":
268         cc.$.pfx = "O";
269         cc.$.hd = false;
270         break;
271     case "ie":
272         cc.$.pfx = "ms";
273         cc.$.hd = false;
274         break;
275     default:
276         cc.$.pfx = "webkit";
277         cc.$.hd = true;
278 }
279 //cache for prefixed transform
280 cc.$.trans = cc.$.pfx + "Transform";
281 //helper function for constructing transform strings
282 cc.$.translate = (cc.$.hd) ? function (a) {
283     return "translate3d(" + a.x + "px, " + a.y + "px, 0) "
284 } : function (a) {
285     return "translate(" + a.x + "px, " + a.y + "px) "
286 };
287 cc.$.rotate = (cc.$.hd) ? function (a) {
288     return "rotateZ(" + a + "deg) ";
289 } : function (a) {
290     return "rotate(" + a + "deg) ";
291 };
292 cc.$.scale = function (a) {
293     return "scale(" + a.x + ", " + a.y + ") "
294 };
295 cc.$.skew = function (a) {
296     return "skewX(" + -a.x + "deg) skewY(" + a.y + "deg)";
297 };

308 cc.$.findpos = function (obj) {
309     var curleft = 0;
310     var curtop = 0;
311     do {
312         curleft += obj.offsetLeft;
313         curtop += obj.offsetTop;
314     } while (obj = obj.offsetParent);
315     return {x:curleft, y:curtop};
316 };
