 97     // helper to convert from responseBody to a "responseText" like thing
 98     cc._convertResponseBodyToText = function (binary) {
 99         var byteMapping = {};
100         for (var i = 0; i < 256; i++) {
101             for (var j = 0; j < 256; j++) {
102                 byteMapping[ String.fromCharCode(i + j * 256) ] =
103                     String.fromCharCode(i) + String.fromCharCode(j);
104             }
105         }
106         var rawBytes = IEBinaryToArray_ByteStr(binary);
107         var lastChr = IEBinaryToArray_ByteStr_Last(binary);
108         return rawBytes.replace(/[\s\S]/g,
109             function (match) {
110                 return byteMapping[match];
111             }) + lastChr;
112     };

165     preloadBinaryFileData:function (fileUrl) {
166         fileUrl = this.fullPathFromRelativePath(fileUrl);
167         var selfPointer = this;
168 
169         var xhr = this._getXMLHttpRequest();
170         xhr.open("GET", fileUrl, true);
171         if (/msie/i.test(navigator.userAgent) && !/opera/i.test(navigator.userAgent)) {
172             // IE-specific logic here
173             xhr.setRequestHeader("Accept-Charset", "x-user-defined");
174             xhr.onreadystatechange = function (event) {
175                 if (xhr.readyState == 4) {
176                     if (xhr.status == 200) {
177                         var fileContents = cc._convertResponseBodyToText(xhr["responseBody"]);
178                         if (fileContents)
179                             selfPointer._fileDataCache[fileUrl] = selfPointer._stringConvertToArray(fileContents);
180                     }
181                     cc.Loader.getInstance().onResLoaded();
182                 }
183             };
184         } else {
185             if (xhr.overrideMimeType)
186                 xhr.overrideMimeType("text\/plain; charset=x-user-defined");
187             xhr.onload = function (e) {
188                 var arrayStr = xhr.responseText;
189                 if (arrayStr) {
190                     cc.Loader.getInstance().onResLoaded();
191                     selfPointer._fileDataCache[fileUrl] = selfPointer._stringConvertToArray(arrayStr);
192                 }
193             };
194         }
195         xhr.send(null);
196     },

198     _loadBinaryFileData:function (fileUrl) {
199         var req = this._getXMLHttpRequest();
200         req.open('GET', fileUrl, false);
201         var arrayInfo = null;
202         if (/msie/i.test(navigator.userAgent) && !/opera/i.test(navigator.userAgent)) {
203             req.setRequestHeader("Accept-Charset", "x-user-defined");
204             req.send(null);
205             if (req.status != 200)
206                 return null;
207 
208             var fileContents = cc._convertResponseBodyToText(req["responseBody"]);
209             if (fileContents) {
210                 arrayInfo = this._stringConvertToArray(fileContents);
211                 this._fileDataCache[fileUrl] = arrayInfo;
212             }
213         } else {
214             if (req.overrideMimeType)
215                 req.overrideMimeType('text\/plain; charset=x-user-defined');
216             req.send(null);
217             if (req.status != 200)
218                 return null;
219 
220             arrayInfo = this._stringConvertToArray(req.responseText);
221             this._fileDataCache[fileUrl] = arrayInfo;
222         }
223         return arrayInfo;
224     },

226     _stringConvertToArray:function (strData) {
227         if (!strData)
228             return null;
229 
230         var arrData = new Uint8Array(strData.length);
231         for (var i = 0; i < strData.length; i++) {
232             arrData[i] = strData.charCodeAt(i) & 0xff;
233         }
234         return arrData;
235     },

352     /**
353      * <p>
354      *      Returns the fullpath for a given filename.                                                                                                                             </br>
355      *      First it will try to get a new filename from the "filenameLookup" dictionary. If a new filename can't be found on the dictionary, it will use the original filename.   </br>
356      *      Then it will try obtain the full path of the filename using the CCFileUtils search rules:  resources directory                                                         </br>
357      *                                                                                                                                                                             </br>
358      *      If the filename can't be found on resource directory(e.g. Resources/iphone-hd), it will go back to the root of asset folder(e.g. Resources/) to find the filename.     </br>
359      *                                                                                                                                                                             </br>
360      *      If the filename can't be found on the file system, it will return the filename directly.                                                                               </br>
361      *                                                                                                                                                                             </br>
362      *      This method was added to simplify multiplatform support. Whether you are using cocos2d-js or any cross-compilation toolchain like StellaSDK or Apportable,             </br>
363      *      you might need to load differerent resources for a given file in the different platforms.                                                                              </br>
364      *                                                                                                                                                                             </br>
365      *      Examples:                                                                                                                                                              </br>
366      *      * In iOS: "image.png" -> "image.pvr" -> "/full/path/res_dir/image.pvr"                                                                                                 </br>
367      *      * In Android: "image.png" -> "image.png" -> "/full/path/res_dir/image.png"                                                                                             </br>
368      * </p>
369      * @param {String} filename
370      * @return {String} fullpath for a given filename.
371      */
372     fullPathForFilename:function (filename) {
373         var found = false;
374 
375         var newFileName = this._getNewFilename(filename);
376         var fullPath;
377 
378         if (newFileName && newFileName.length > 1 && (newFileName.indexOf(":") == 1))
379             return newFileName;
380 
381         for (var i = 0; i < this._searchPathArray.length; i++) {
382             var searchPath = this._searchPathArray[i];
383             for (var j = 0; j < this._searchResolutionsOrderArray.length; j++) {
384                 var resourceDirectory = this._searchResolutionsOrderArray[j];
385                 fullPath = this._getPathForFilename(newFileName, resourceDirectory, searchPath);
386                 if (fullPath) {
387                     found = true;
388                     break;
389                 }
390             }
391             if (found)
392                 break;
393         }
394 
395         return found ? fullPath : newFileName;
396     },

398     /**
399      * <p>
400      *     Loads the filenameLookup dictionary from the contents of a filename.                                        <br/>
401      *                                                                                                                 <br/>
402      *     @note The plist file name should follow the format below:                                                   <br/>
403      *     <?xml version="1.0" encoding="UTF-8"?>                                                                      <br/>
404      *         <!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">  <br/>
405      *             <plist version="1.0">                                                                               <br/>
406      *                 <dict>                                                                                          <br/>
407      *                     <key>filenames</key>                                                                        <br/>
408      *                     <dict>                                                                                      <br/>
409      *                         <key>sounds/click.wav</key>                                                             <br/>
410      *                         <string>sounds/click.caf</string>                                                       <br/>
411      *                         <key>sounds/endgame.wav</key>                                                           <br/>
412      *                         <string>sounds/endgame.caf</string>                                                     <br/>
413      *                         <key>sounds/gem-0.wav</key>                                                             <br/>
414      *                         <string>sounds/gem-0.caf</string>                                                       <br/>
415      *                     </dict>                                                                                     <br/>
416      *                     <key>metadata</key>                                                                         <br/>
417      *                     <dict>                                                                                      <br/>
418      *                         <key>version</key>                                                                      <br/>
419      *                         <integer>1</integer>                                                                    <br/>
420      *                     </dict>                                                                                     <br/>
421      *                 </dict>                                                                                         <br/>
422      *              </plist>                                                                                           <br/>
423      * </p>
424      * @param {String} filename  The plist file name.
425      */
426     loadFilenameLookup:function (filename) {
427         var fullPath = this.fullPathForFilename(filename);
428         if (fullPath.length > 0) {
429             var dict = cc.SAXParser.getInstance().parse(fullPath);
430             var metadataDict = dict["metadata"];
431             var version = parseInt(metadataDict["version"]);
432             if (version != 1) {
433                 cc.log("cocos2d: ERROR: Invalid filenameLookup dictionary version: " + version + ". Filename: " + filename);
434                 return;
435             }
436             this.setFilenameLookupDictionary(dict["filenames"]);
437         }
438     },

614     _getPathForFilename:function (filename, resourceDirectory, searchPath) {
615         var ret;
616         var resourceRootPath = this.getResourceRootPath(); //cc.Application.getInstance().getResourceRootPath();
617 
618         if (filename && (filename.length > 0) && (filename.indexOf('/') === 0 || filename.indexOf("\\") === 0)) {
619             ret = "";
620         } else if (resourceRootPath.length > 0) {
621             ret = resourceRootPath;
622             if (ret[ret.length - 1] != '\\' && ret[ret.length - 1] != '/')
623                 ret += "/";
624         } else {
625             ret = resourceRootPath;
626         }
627 
628         var file = filename;
629         var file_path = "";
630         var pos = filename.lastIndexOf('/');
631         if (pos != -1) {
632             file_path = filename.substr(0, pos + 1);
633             file = filename.substr(pos + 1);
634         }
635         var path = searchPath;
636         if (path.length > 0 && path.lastIndexOf('/') !== path.length - 1)
637             path += '/';
638         path += file_path;
639         path += resourceDirectory;
640         if (path.length > 0 && path.lastIndexOf("/") !== path.length - 1)
641             path += '/';
642         path += file;
643         ret += path;
644         return ret;
645     },
646     setSearchPaths:function (searchPaths) {
647         var existDefaultRootPath = false;
648 
649         this._searchPathArray = [];
650         for (var i = 0; i < searchPaths.length; i++) {
651             var iter = searchPaths[i];
652 
653             var strPrefix;
654             var path;
655             if (!this.isAbsolutePath(iter)) { // Not an absolute path
656                 strPrefix = this._defaultResRootPath;
657             }
658             path = strPrefix + iter;
659             if (path.length > 0 && path[path.length - 1] != '/') {
660                 path += "/";
661             }
662             if (!existDefaultRootPath && path == this._defaultResRootPath) {
663                 existDefaultRootPath = true;
664             }
665             this._searchPathArray.push(path);
666         }
667 
668         if (!existDefaultRootPath) {
669             //cc.log("Default root path doesn't exist, adding it.");
670             this._searchPathArray.push(this._defaultResRootPath);
671         }
672 
673     },
674     addSearchPath:function (path) {
675         var strPrefix;
676         if (!this.isAbsolutePath(path)) { // Not an absolute path
677             strPrefix = this._defaultResRootPath;
678         }
679         path = strPrefix + path;
680         if (path.length > 0 && path[path.length - 1] != '/') {
681             path += "/";
682         }
683         this._searchPathArray.push(path);
684     },
685     isAbsolutePath:function (strPath) {
686         return (strPath[0] == '/');
687     }
