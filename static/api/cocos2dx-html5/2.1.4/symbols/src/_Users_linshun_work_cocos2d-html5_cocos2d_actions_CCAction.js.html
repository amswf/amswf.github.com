 33 /**
 34  * Base class for cc.Action objects.
 35  * @class
 36  * @extends cc.Class
 37  */
 38 cc.Action = cc.Class.extend(/** @lends cc.Action# */{
 39     //***********variables*************
 40     _originalTarget:null,
 41 
 42     /** The "target".
 43      The target will be set with the 'startWithTarget' method.
 44      When the 'stop' method is called, target will be set to nil.
 45      The target is 'assigned', it is not 'retained'.
 46      */
 47     _target:null,
 48     _tag:cc.ACTION_TAG_INVALID,
 49 
 50     //**************Public Functions***********
 51     /**
 52      * @return {String}
 53      */
 54     description:function () {
 55         return "<cc.Action | Tag = " + this._tag + ">";
 56     },
 57 
 58     /**
 59      * to copy object with deep copy.
 60      * @param {object} zone
 61      * @return {object}
 62      */
 63     copyWithZone:function (zone) {
 64         return this.copy();
 65     },
 66 
 67     /**
 68      * to copy object with deep copy.
 69      * @return {object}
 70      */
 71     copy:function () {
 72         return cc.clone(this);
 73     },
 74 
 75     /**
 76      * return true if the action has finished
 77      * @return {Boolean}
 78      */
 79     isDone:function () {
 80         return true;
 81     },
 82 
 83     /**
 84      * called before the action start. It will also set the target.
 85      * @param {cc.Node} target
 86      */
 87     startWithTarget:function (target) {
 88         this._originalTarget = target;
 89         this._target = target;
 90     },
 91 
 92     /**
 93      * called after the action has finished. It will set the 'target' to nil.
 94      * IMPORTANT: You should never call "action stop" manually. Instead, use: "target.stopAction(action);"
 95      */
 96     stop:function () {
 97         this._target = null;
 98     },
 99     /** called every frame with it's delta time. DON'T override unless you know what you are doing.
100      *
101      * @param {Number} dt
102      */
103 
104     step:function (dt) {
105         cc.log("[Action step]. override me");
106     },
107 
108     /**
109      <p>called once per frame. time a value between 0 and 1  </P>
110 
111      <p>For example:  <br/>
112      - 0 means that the action just started <br/>
113      - 0.5 means that the action is in the middle<br/>
114      - 1 means that the action is over </P>
115      * @param {Number}  time
116      */
117     update:function (time) {
118         cc.log("[Action update]. override me");
119     },
120 
121     /**
122      *
123      * @return {cc.Node}
124      */
125     getTarget:function () {
126         return this._target;
127     },
128 
129     /** The action will modify the target properties.
130      *
131      * @param {cc.Node} target
132      */
133     setTarget:function (target) {
134         this._target = target;
135     },
136 
137     /**
138      *
139      * @return {cc.Node}
140      */
141     getOriginalTarget:function () {
142         return this._originalTarget;
143     },
144 
145     /** Set the original target, since target can be nil. <br/>
146      * Is the target that were used to run the action.  <br/>
147      * Unless you are doing something complex, like cc.ActionManager, you should NOT call this method. <br/>
148      * The target is 'assigned', it is not 'retained'. <br/>
149      * @param {cc.Node} originalTarget
150      */
151     setOriginalTarget:function (originalTarget) {
152         this._originalTarget = originalTarget;
153     },
154 
155     /**
156      *
157      * @return {Number}
158      */
159     getTag:function () {
160         return this._tag;
161     },
162 
163     /**
164      *
165      * @param {Number} tag
166      */
167     setTag:function (tag) {
168         this._tag = tag;
169     },
170     /**
171      * Currently JavaScript Bindigns (JSB), in some cases, needs to use retain and release. This is a bug in JSB,
172      * and the ugly workaround is to use retain/release. So, these 2 methods were added to be compatible with JSB.
173      * This is a hack, and should be removed once JSB fixes the retain/release bug
174      */
175     retain:function () {
176     },
177     release:function () {
178     }
179 });

181  * @returns {cc.Action}
182  * @example
183  * // example
184  * var action = cc.Action.create();
185  */
186 cc.Action.create = function () {
187     return new cc.Action();
188 };

232 /**
233  * Changes the speed of an action, making it take longer (speed>1)
234  * or less (speed<1) time. <br/>
235  * Useful to simulate 'slow motion' or 'fast forward' effect.
236  * @warning This action can't be Sequenceable because it is not an cc.IntervalAction
237  * @class
238  * @extends cc.Action
239  */
240 cc.Speed = cc.Action.extend(/** @lends cc.Speed# */{
241     _speed:0.0,
242     _innerAction:null,
243 
244     /**
245      * @return {Number}
246      */
247     getSpeed:function () {
248         return this._speed;
249     },
250 
251     /** alter the speed of the inner function in runtime
252      * @param {Number} speed
253      */
254     setSpeed:function (speed) {
255         this._speed = speed;
256     },
257 
258     /** initializes the action
259      * @param {cc.ActionInterval} action
260      * @param {Number} speed
261      * @return {Boolean}
262      */
263     initWithAction:function (action, speed) {
264         cc.Assert(action != null, "");
265         this._innerAction = action;
266         this._speed = speed;
267         return true;
268     },
269 
270     /**
271      * @param {cc.Node} target
272      */
273     startWithTarget:function (target) {
274         cc.Action.prototype.startWithTarget.call(this, target);
275         this._innerAction.startWithTarget(target);
276     },
277 
278     /**
279      *  Stop the action
280      */
281     stop:function () {
282         this._innerAction.stop();
283         cc.Action.prototype.stop.call(this);
284     },
285 
286     /**
287      * @param {Number} dt
288      */
289     step:function (dt) {
290         this._innerAction.step(dt * this._speed);
291     },
292 
293     /**
294      * @return {Boolean}
295      */
296     isDone:function () {
297         return this._innerAction.isDone();
298     },
299 
300     /**
301      * @return {cc.ActionInterval}
302      */
303     reverse:function () {
304         return (cc.Speed.create(this._innerAction.reverse(), this._speed));
305     },
306 
307     /**
308      *
309      * @param {cc.ActionInterval} action
310      */
311     setInnerAction:function (action) {
312         if (this._innerAction != action) {
313             this._innerAction = action;
314         }
315     },
316 
317     /**
318      *
319      * @return {cc.ActionInterval}
320      */
321     getInnerAction:function () {
322         return this._innerAction;
323     }
324 });
325 /** creates the action
326  *
327  * @param {cc.ActionInterval} action
328  * @param {Number} speed
329  * @return {cc.Speed}
330  */
331 cc.Speed.create = function (action, speed) {
332     var ret = new cc.Speed();
333     if (ret && ret.initWithAction(action, speed)) {
334         return ret;
335     }
336     return null;
337 };

339 /**
340  * cc.Follow is an action that "follows" a node.
341 
342  * @example
343  * //example
344  * //Instead of using cc.Camera as a "follower", use this action instead.
345  * layer.runAction(cc.Follow.actionWithTarget(hero));
346 
347  * @class
348  * @extends cc.Action
349  */
350 cc.Follow = cc.Action.extend(/** @lends cc.Follow# */{
351     /**
352      * @return {Boolean}
353      */
354     isBoundarySet:function () {
355         return this._boundarySet;
356     },
357 
358     /** alter behavior - turn on/off boundary
359      * @param {Boolean} value
360      */
361     setBoudarySet:function (value) {
362         this._boundarySet = value;
363     },
364 
365     /** initializes the action
366      * initializes the action with a set boundary
367      * @param {cc.Node} followedNode
368      * @param {cc.Rect} rect
369      * @return {Boolean}
370      */
371     initWithTarget:function (followedNode, rect) {
372         cc.Assert(followedNode != null, "");
373 
374         rect = rect || cc.RectZero();
375         this._followedNode = followedNode;
376 
377         this._boundarySet = !cc.rectEqualToRect(rect, cc.RectZero());
378 
379         this._boundaryFullyCovered = false;
380 
381         var winSize = cc.Director.getInstance().getWinSize();
382         this._fullScreenSize = cc.p(winSize.width, winSize.height);
383         this._halfScreenSize = cc.pMult(this._fullScreenSize, 0.5);
384 
385         if (this._boundarySet) {
386             this.leftBoundary = -((rect.x + rect.width) - this._fullScreenSize.x);
387             this.rightBoundary = -rect.x;
388             this.topBoundary = -rect.y;
389             this.bottomBoundary = -((rect.y + rect.height) - this._fullScreenSize.y);
390 
391             if (this.rightBoundary < this.leftBoundary) {
392                 // screen width is larger than world's boundary width
393                 //set both in the middle of the world
394                 this.rightBoundary = this.leftBoundary = (this.leftBoundary + this.rightBoundary) / 2;
395             }
396             if (this.topBoundary < this.bottomBoundary) {
397                 // screen width is larger than world's boundary width
398                 //set both in the middle of the world
399                 this.topBoundary = this.bottomBoundary = (this.topBoundary + this.bottomBoundary) / 2;
400             }
401 
402             if ((this.topBoundary == this.bottomBoundary) && (this.leftBoundary == this.rightBoundary)) {
403                 this._boundaryFullyCovered = true;
404             }
405         }
406         return true;
407     },
408 
409     /**
410      * @param {Number} dt
411      */
412     step:function (dt) {
413         if (this._boundarySet) {
414             // whole map fits inside a single screen, no need to modify the position - unless map boundaries are increased
415             if (this._boundaryFullyCovered)
416                 return;
417 
418             var tempPos = cc.pSub(this._halfScreenSize, this._followedNode.getPosition());
419 
420             this._target.setPosition(cc.p(cc.clampf(tempPos.x, this.leftBoundary, this.rightBoundary),
421                 cc.clampf(tempPos.y, this.bottomBoundary, this.topBoundary)));
422         } else {
423             this._target.setPosition(cc.pSub(this._halfScreenSize, this._followedNode.getPosition()));
424         }
425     },
426 
427     /**
428      * @return {Boolean}
429      */
430     isDone:function () {
431         return ( !this._followedNode.isRunning() );
432     },
433 
434     /**
435      * Stop the action.
436      */
437     stop:function () {
438         this._target = null;
439         cc.Action.prototype.stop.call(this);
440     },
441 
442     // node to follow
443     _followedNode:null,
444     // whether camera should be limited to certain area
445     _boundarySet:false,
446     // if screen size is bigger than the boundary - update not needed
447     _boundaryFullyCovered:false,
448     // fast access to the screen dimensions
449     _halfScreenSize:null,
450     _fullScreenSize:null,
451 
452     /** world leftBoundary
453      * @Type {Number}
454      */
455     leftBoundary:0.0,
456     /** world rightBoundary
457      * @Type Number
458      */
459     rightBoundary:0.0,
460     /** world topBoundary
461      * @Type Number
462      */
463     topBoundary:0.0,
464     /** world bottomBoundary
465      * @Type {Number}
466      */
467     bottomBoundary:0.0
468 });
469 /** creates the action with a set boundary <br/>
470  * creates the action with no boundary set
471  * @param {cc.Node} followedNode
472  * @param {cc.Rect} rect
473  * @return {cc.Follow|Null} returns the cc.Follow object on success
474  * @example
475  * // example
476  * // creates the action with a set boundary
477  * var sprite = cc.Sprite.create("spriteFileName");
478  * var followAction = cc.Follow.create(sprite, cc.rect(0, 0, s.width * 2 - 100, s.height));
479  * this.runAction(followAction);
480  *
481  * // creates the action with no boundary set
482  * var sprite = cc.Sprite.create("spriteFileName");
483  * var followAction = cc.Follow.create(sprite);
484  * this.runAction(followAction);
485  */
486 cc.Follow.create = function (followedNode, rect) {
487     rect = rect || new cc.RectZero();
488     var ret = new cc.Follow();
489     if (rect != null && ret && ret.initWithTarget(followedNode, rect)) {
490         return ret;
491     }
492     else if (ret && ret.initWithTarget(followedNode)) {
493         return ret;
494     }
495     return null;
496 };
